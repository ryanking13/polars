13fn sum_between_offsets<T, S>(values: &[T], offset: &[i64]) -> Vec<S>
14where
15    T: NativeType + ToPrimitive,
16    S: NumCast + std::iter::Sum,
17{
18    offset
19        .windows(2)
20        .map(|w| {
21            values
22                .get(w[0] as usize..w[1] as usize)
23                .map(sum_slice)
24                .unwrap_or(S::from(0).unwrap())
25        })
26        .collect()
27}

29fn dispatch_sum<T, S>(arr: &dyn Array, offsets: &[i64], validity: Option<&Bitmap>) -> ArrayRef
30where
31    T: NativeType + ToPrimitive,
32    S: NativeType + NumCast + std::iter::Sum,
33{
34    let values = arr.as_any().downcast_ref::<PrimitiveArray<T>>().unwrap();
35    let values = values.values().as_slice();
36    Box::new(PrimitiveArray::from_data_default(
37        sum_between_offsets::<_, S>(values, offsets).into(),
38        validity.cloned(),
39    )) as ArrayRef
40}

42pub(super) fn sum_list_numerical(ca: &ListChunked, inner_type: &DataType) -> Series {
43    use DataType::*;
44    let chunks = ca
45        .downcast_iter()
46        .map(|arr| {
47            let offsets = arr.offsets().as_slice();
48            let values = arr.values().as_ref();
49
50            match inner_type {
51                Int8 => dispatch_sum::<i8, i64>(values, offsets, arr.validity()),
52                Int16 => dispatch_sum::<i16, i64>(values, offsets, arr.validity()),
53                Int32 => dispatch_sum::<i32, i32>(values, offsets, arr.validity()),
54                Int64 => dispatch_sum::<i64, i64>(values, offsets, arr.validity()),
55                Int128 => dispatch_sum::<i128, i128>(values, offsets, arr.validity()),
56                UInt8 => dispatch_sum::<u8, i64>(values, offsets, arr.validity()),
57                UInt16 => dispatch_sum::<u16, i64>(values, offsets, arr.validity()),
58                UInt32 => dispatch_sum::<u32, u32>(values, offsets, arr.validity()),
59                UInt64 => dispatch_sum::<u64, u64>(values, offsets, arr.validity()),
60                Float32 => dispatch_sum::<f32, f32>(values, offsets, arr.validity()),
61                Float64 => dispatch_sum::<f64, f64>(values, offsets, arr.validity()),
62                _ => unimplemented!(),
63            }
64        })
65        .collect::<Vec<_>>();
66
67    Series::try_from((ca.name().clone(), chunks)).unwrap()
68}

70pub(super) fn sum_with_nulls(ca: &ListChunked, inner_dtype: &DataType) -> PolarsResult<Series> {
71    use DataType::*;
72    // TODO: add fast path for smaller ints?
73    let mut out = match inner_dtype {
74        Boolean => {
75            let out: IdxCa =
76                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<IdxSize>().unwrap()));
77            out.into_series()
78        },
79        UInt32 => {
80            let out: UInt32Chunked =
81                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<u32>().unwrap()));
82            out.into_series()
83        },
84        UInt64 => {
85            let out: UInt64Chunked =
86                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<u64>().unwrap()));
87            out.into_series()
88        },
89        Int32 => {
90            let out: Int32Chunked =
91                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i32>().unwrap()));
92            out.into_series()
93        },
94        Int64 => {
95            let out: Int64Chunked =
96                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<i64>().unwrap()));
97            out.into_series()
98        },
99        Float32 => {
100            let out: Float32Chunked =
101                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<f32>().unwrap()));
102            out.into_series()
103        },
104        Float64 => {
105            let out: Float64Chunked =
106                ca.apply_amortized_generic(|s| s.map(|s| s.as_ref().sum::<f64>().unwrap()));
107            out.into_series()
108        },
109        // slowest sum_as_series path
110        dt => ca
111            .try_apply_amortized(|s| {
112                s.as_ref()
113                    .sum_reduce()
114                    .map(|sc| sc.into_series(PlSmallStr::EMPTY))
115            })?
116            .explode(false)
117            .unwrap()
118            .into_series()
119            .cast(dt)?,
120    };
121    out.rename(ca.name().clone());
122    Ok(out)
123}

125fn mean_between_offsets<T, S>(values: &[T], offset: &[i64]) -> PrimitiveArray<S>
126where
127    T: NativeType + ToPrimitive,
128    S: NativeType + NumCast + std::iter::Sum + Div<Output = S>,
129{
130    offset
131        .windows(2)
132        .map(|w| {
133            values
134                .get(w[0] as usize..w[1] as usize)
135                .filter(|sl| !sl.is_empty())
136                .map(|sl| sum_slice::<_, S>(sl) / NumCast::from(sl.len()).unwrap())
137        })
138        .collect()
139}

141fn dispatch_mean<T, S>(arr: &dyn Array, offsets: &[i64], validity: Option<&Bitmap>) -> ArrayRef
142where
143    T: NativeType + ToPrimitive,
144    S: NativeType + NumCast + std::iter::Sum + Div<Output = S>,
145{
146    let values = arr.as_any().downcast_ref::<PrimitiveArray<T>>().unwrap();
147    let values = values.values().as_slice();
148    let out = mean_between_offsets::<_, S>(values, offsets);
149    let new_validity = combine_validities_and(out.validity(), validity);
150    out.with_validity(new_validity).to_boxed()
151}

181pub(super) fn mean_with_nulls(ca: &ListChunked) -> Series {
182    match ca.inner_dtype() {
183        DataType::Float32 => {
184            let out: Float32Chunked = ca
185                .apply_amortized_generic(|s| s.and_then(|s| s.as_ref().mean().map(|v| v as f32)))
186                .with_name(ca.name().clone());
187            out.into_series()
188        },
189        #[cfg(feature = "dtype-datetime")]
190        DataType::Date => {
191            let out: Int64Chunked = ca
192                .apply_amortized_generic(|s| {
193                    s.and_then(|s| s.as_ref().mean().map(|v| (v * (US_IN_DAY as f64)) as i64))
194                })
195                .with_name(ca.name().clone());
196            out.into_datetime(TimeUnit::Microseconds, None)
197                .into_series()
198        },
199        dt if dt.is_temporal() => {
200            let out: Int64Chunked = ca
201                .apply_amortized_generic(|s| s.and_then(|s| s.as_ref().mean().map(|v| v as i64)))
202                .with_name(ca.name().clone());
203            out.cast(dt).unwrap()
204        },
205        _ => {
206            let out: Float64Chunked = ca
207                .apply_amortized_generic(|s| s.and_then(|s| s.as_ref().mean()))
208                .with_name(ca.name().clone());
209            out.into_series()
210        },
211    }
